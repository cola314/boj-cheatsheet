1. Top-Down (메모이제이션) vs Bottom-Up

방식 특징 패턴
Top-Down 재귀 + memo 배열 if (ret != -1) return ret;
Bottom-Up 반복문 테이블 채우기 dp[i] = dp[i-1] + ...

memset 값 선택 가이드

값 memset 인수 용도

3. 1D DP — 최소 연산 (1463 1로 만들기)

int memo[1000010];

int f(int n) {
    if (memo[n] == 987654321) {               // 미계산 확인
        if (n % 3 == 0) memo[n] = min(memo[n], f(n/3) + 1);
        if (n % 2 == 0) memo[n] = min(memo[n], f(n/2) + 1);
        memo[n] = min(memo[n], f(n-1) + 1);
    }
    return memo[n];
}

// 초기화: INF로 채우고 기저값 설정
for (int i = 1; i <= 1000000; i++) memo[i] = 987654321;
memo[1] = 0;

4. Top-Down — 계단 오르기 (2579)

int stair[310], memo[310];

int f(int n) {
    if (n < 0) return 0;
    // 0 초기화 활용: memo[n]==0 이면 미계산 (값이 0일 수 없는 문제)
    if (!memo[n])
        memo[n] = stair[n] + max(f(n-2), f(n-3) + stair[n-1]);
    return memo[n];
}

주의: !memo[n] 패턴은 dp값이 0이 될 수 없을 때만 사용 가능. 범용은 memset -1 권장.

6. LIS — 최장 증가 부분수열 (11053)

int dp[1010], result = 1;
vector<int> V;

// O(N^2) 기본 LIS
for (int i = 0; i < n; i++) dp[i] = 1;
for (int i = 0; i < n; i++) {
    for (int j = 0; j < i; j++) {
        if (V[j] < V[i]) {
            dp[i] = max(dp[i], dp[j] + 1);
            result = max(result, dp[i]);
        }
    }
}

O(N log N) LIS는 lower_bound + patience sorting 사용.

7. LCS — 최장 공통 부분수열 (9251)

string s1, s2;
int arr[1010][1010];

for (int i = 1; i <= s1.size(); i++) {
    for (int j = 1; j <= s2.size(); j++) {
        if (s1[i-1] == s2[j-1])
            arr[i][j] = max(arr[i][j-1], arr[i-1][j-1] + 1);
        else
            arr[i][j] = max(arr[i][j-1], arr[i-1][j]);
    }
}
cout << arr[s1.size()][s2.size()];

LCS 점화식 요약

s1[i-1] == s2[j-1]:  dp[i][j] = dp[i-1][j-1] + 1
otherwise:            dp[i][j] = max(dp[i-1][j], dp[i][j-1])

8. 0/1 배낭 (12865)

int dp[101][100010];
vector<pair<int,int>> V;  // {무게, 가치}

// Top-Down
int f(int item, int weight) {
    if (item == 0) return 0;
    if (dp[item][weight] == -1) {
        if (weight - V[item].first >= 0)
            dp[item][weight] = f(item-1, weight - V[item].first) + V[item].second;
        dp[item][weight] = max(dp[item][weight], f(item-1, weight));
    }
    return dp[item][weight];
}

// 초기화
for (int i = 0; i <= 100; i++) memset(dp[i], -1, sizeof(dp[i]));

Bottom-Up 배낭 (더 일반적)

// dp[w] = 무게 w 이하에서 최대 가치
for (int i = 1; i <= n; i++)
    for (int w = K; w >= V[i].first; w--)  // 역순! 중복 방지
        dp[w] = max(dp[w], dp[w - V[i].first] + V[i].second);

9. 비트마스크 DP — TSP (2098)

int n, w[20][20];
int memo[16][(1 << 16)];   // [현재노드][방문비트마스크]

int f(int mask, int pos) {
    if (mask == (1 << n) - 1)    // 모두 방문
        return w[pos][0];         // 시작점으로 복귀
    if (memo[pos][mask] != -1) return memo[pos][mask];

    int re = INF;
    for (int i = 0; i < n; i++) {
        if (~mask & (1 << i) && w[pos][i] != INF) {  // 미방문 & 경로 존재
            re = min(re, f(mask | (1 << i), i) + w[pos][i]);
        }
    }
    return memo[pos][mask] = re;
}

// 초기 호출: 0번 노드에서 출발, 0번 방문 표시
f(1, 0);   // mask=1 = 0번 노드 방문

비트마스크 연산 정리

연산 코드 의미
i번 방문 mask \| (1 << i) i번 비트 set
i번 미방문 확인 ~mask & (1 << i) i번 비트 0인지
전체 방문 확인 mask == (1<<n)-1 모든 비트 1

10. 구간 DP (11066 파일합치기)

int arr[510], memo[510][510], sum[510];

int f(int a, int b) {
    if (a == b) return 0;                    // 기저: 파일 1개 = 비용 0
    if (memo[a][b] != -1) return memo[a][b];
    int re = INF;
    for (int i = a; i < b; i++) {
        re = min(re, f(a, i) + f(i+1, b) + sum[b] - sum[a-1]);
    }                                        // 구간 합: prefix sum 활용
    return memo[a][b] = re;
}

// prefix sum 초기화
for (int i = 1; i <= n; i++) {
    cin >> arr[i];
    sum[i] = sum[i-1] + arr[i];
    memset(memo[i], -1, sizeof(memo[i]));   // 테스트케이스마다 초기화
}

구간 DP 점화식: dp[a][b] = min(dp[a][k] + dp[k+1][b] + cost(a,b))

센티널(기저값) 패턴 모음

문제 유형 기저 조건 코드
1로 만들기 n==1 memo[1] = 0
계단 오르기 n < 0 if (n < 0) return 0
배낭 item==0 if (item == 0) return 0

패턴 빈도 & 주의점

  • INF 정의: 226회 — 987654321 또는 1000000010 자주 사용
  • memo 배열: 42회 — 전역 선언으로 0 초기화 활용 또는 memset -1
  • 레퍼런스 트릭: int& ret = memo[...] — Top-Down의 핵심
  • 오버플로우: INF + INF가 음수 되지 않게 주의 (long long 사용 or 조건 체크)
  • 구간 DP 초기화: 테스트케이스마다 memset 재호출 필요
  • 비트마스크: n≤20일 때만 (1<<n) 사용 — n=20이면 배열 크기 1MB 초과 주의
